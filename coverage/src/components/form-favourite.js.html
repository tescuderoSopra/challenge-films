    _dropZoneEvents() {
        const dropZone = this.shadowRoot.getElementById('backdrop_path');
        if (dropZone) {
            dropZone.addEventListener('dragover', (e) => {
                e.stopPropagation();

                e.dataTransfer.dropEffect = 'copy';
            });
            dropZone.addEventListener('drop', (e) => {
                e.stopPropagation();
                e.preventDefault();
                const files = e.dataTransfer.files;
